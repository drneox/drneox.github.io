        else:
            return False

     def peek(self):
        return self.queue[0]

        self.right = None
        self.data = data

    def create_node(self, data):
        '''
        we create node as a list according to the letters of the alphabet.
        '''
        if data < self.data:

                self.left.create_node(data)
        else:
            if self.right is None:
                self.right = Tree(data)

                self.right.create_node(data)
